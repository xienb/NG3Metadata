2、执行以下sql语句建表

create table hr_employee
(
   id varchar(100) primary key,
   name varchar(200),
   sex varchar(1),
   age int,
   birthday date,
   province varchar(10),
   city varchar(10),
   email varchar(200) 
)

create table hr_emp_positoninfo
(
    id varchar(100) primary key,
	mainid varchar(100) ,
	workdt date ,
	department varchar(100) ,
	station varchar(100) ,
	postion varchar(100) ,
	level int 
)
